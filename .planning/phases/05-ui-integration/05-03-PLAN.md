  <action>Add methods to wire UI controls to physics simulation:

1. In GLHost.cs, add SetControlMode method:
   ```csharp
   public void SetControlMode(ControlMode mode)
   {
       if (!_glLoaded) return;
       _physics.Mode = mode;

       // Reset accumulator when pausing/unpausing to prevent time spikes
       if (mode == ControlMode.Manual)
       {
           _physics.ResetAccumulator();
       }
   }
   ```

2. In GLHost.cs, add manual override methods:
   ```csharp
   public void SetManualPosition(Vector3 position)
   {
       if (!_glLoaded) return;
       _physics.ManualPosition = position;
       ApplyManualIfNeeded();
   }

   public void SetManualRotation(Vector3 rotation)
   {
       if (!_glLoaded) return;
       _physics.ManualRotation = rotation;
       ApplyManualIfNeeded();
   }

   private void ApplyManualIfNeeded()
   {
       if (_physics.Mode == ControlMode.Manual && _particleData != null)
       {
           _physics.ApplyManualOverrides(_particleData, _physics.ManualPosition, _physics.ManualRotation);
           UploadParticlesToGPU(_particleData);
       }
   }
   ```

3. In PhysicsSimulator.cs, implement ApplyManualOverrides:
   ```csharp
   public void ApplyManualOverrides(ParticleData[] particles, Vector3 position, Vector3 rotation)
   {
       if (particles == null) return;

       // For now: simple position offset applied to all particles
       // In Phase 6, rotation will be properly applied
       for (int i = 0; i < particles.Length; i++)
       {
           // Store original relative positions and apply offset
           // This is a simplified implementation - full rotation in Phase 6
           particles[i].Position += position * 0.1f; // Scale factor for sensitivity
       }
   }
   ```
